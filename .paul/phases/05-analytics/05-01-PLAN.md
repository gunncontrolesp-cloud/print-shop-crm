    Create `src/app/dashboard/analytics/page.tsx` as a server component.

    **Data fetching** — aggregate in JavaScript from two Supabase queries (no RPC needed for a small-shop dataset):

    ```ts
    // 1. Paid invoices with customer join (for revenue + top customers)
    const twelveMonthsAgo = new Date()
    twelveMonthsAgo.setMonth(twelveMonthsAgo.getMonth() - 12)
    twelveMonthsAgo.setDate(1)
    twelveMonthsAgo.setHours(0, 0, 0, 0)

    const { data: paidInvoices } = await supabase
      .from('invoices')
      .select('id, amount, created_at, orders(customers(id, name))')
      .eq('status', 'paid')
      .gte('created_at', twelveMonthsAgo.toISOString())

    // 2. All orders for volume trend
    const { data: recentOrders } = await supabase
      .from('orders')
      .select('id, created_at')
      .gte('created_at', twelveMonthsAgo.toISOString())
    ```

    **Build last-12-months labels** — array of `{ label: 'Mar 25', key: '2025-03' }` for the x-axis. Generate from current month going back 12, using `new Date()` arithmetic. Current month is index 11 (rightmost).

    **Aggregate monthly revenue:**
    ```ts
    // Map: '2025-03' → total revenue
    const revenueByMonth: Record<string, number> = {}
    for (const invoice of paidInvoices ?? []) {
      const key = invoice.created_at.slice(0, 7) // 'YYYY-MM'
      revenueByMonth[key] = (revenueByMonth[key] ?? 0) + Number(invoice.amount)
    }
    ```

    **Aggregate top customers:**
    ```ts
    // Map: customerId → { name, revenue, orderCount }
    const customerMap: Record<string, { name: string; revenue: number; orderCount: number }> = {}
    for (const invoice of paidInvoices ?? []) {
      const order = Array.isArray(invoice.orders) ? invoice.orders[0] : invoice.orders
      const customer = Array.isArray(order?.customers) ? order.customers[0] : order?.customers
      if (!customer) continue
      const entry = customerMap[customer.id] ?? { name: customer.name, revenue: 0, orderCount: 0 }
      entry.revenue += Number(invoice.amount)
      entry.orderCount += 1
      customerMap[customer.id] = entry
    }
    const topCustomers = Object.values(customerMap)
      .sort((a, b) => b.revenue - a.revenue)
      .slice(0, 5)
    ```

    **Aggregate monthly order volume:**
    ```ts
    const volumeByMonth: Record<string, number> = {}
    for (const order of recentOrders ?? []) {
      const key = order.created_at.slice(0, 7)
      volumeByMonth[key] = (volumeByMonth[key] ?? 0) + 1
    }
    ```

    **CSS bar chart helper** — no library, pure CSS:
    ```tsx
    function BarChart({ data }: { data: { label: string; value: number }[] }) {
      const max = Math.max(...data.map(d => d.value), 1)
      return (
        <div className="flex items-end gap-1 h-32">
          {data.map((d) => (
            <div key={d.label} className="flex flex-col items-center flex-1 gap-1">
              <div
                className="w-full bg-gray-900 rounded-t min-h-[2px]"
                style={{ height: `${(d.value / max) * 100}%` }}
              />
              <span className="text-[9px] text-gray-400 truncate w-full text-center">
                {d.label}
              </span>
            </div>
          ))}
        </div>
      )
    }
    ```
